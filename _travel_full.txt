function useTpl(idx) {
  const t = templateList[idx]
  newPlan.title = t.title
  newPlan.destination = t.destination
  newPlan.startDate = t.startDate
  newPlan.endDate = t.endDate
  newPlan.budget = t.budget
  newPlan.status = t.status
  newPlan.notes = t.notes
  newPlan.markdown = t.markdown
  createMode.value = "md"
  uni.showToast({ title: "已加载模板", icon: "success" })
}

// ========== 加载 / 持久化 ==========
const STORAGE_KEY = "travelPlans"

const tianjinPlan = {
  id: "static-tianjin",
  title: "天津+北京+滨海新区",
  destination: "天津/北京",
  startDate: "2026-07-16",
  endDate: "2026-07-20",
  budget: "",
  status: "planning",
  notes: "真实出行计划 · 5天4晚 (天津+滨海新区+北京)",
  contentMarkdown: tianjinPlanMarkdown,
  fromStatic: true,
}

function loadPlans() {
  // 1. 从 storage 加载
  let stored = []
  try {
    const raw = uni.getStorageSync(STORAGE_KEY)
    if (raw) stored = JSON.parse(raw)
  } catch (e) { /* empty */ }

  // 2. 天津计划始终在最前面（用户真实计划优先级最高）
  if (!stored.find((p) => p.id === "static-tianjin")) {
    stored.unshift(tianjinPlan)
  }

  // 3. 如果 storage 为空，塞入模板示例
  if (stored.length === 0) {
    stored = [tianjinPlan, ...templateList.map((t, i) => ({
      id: "tpl-" + i,
      ...t,
      fromStatic: false,
    }))]
  }

  plans.value = stored
  uni.setStorageSync(STORAGE_KEY, JSON.stringify(stored))
}

function savePlans() {
  uni.setStorageSync(STORAGE_KEY, JSON.stringify(plans.value))
}

// ========== 操作 ==========
function viewPlan(plan) {
  currentPlan.value = plan
  view.value = "detail"
}

function goBack() {
  if (view.value === "detail") {
    view.value = "list"
    currentPlan.value = null
  } else {
    uni.navigateBack()
  }
}

function openCreate() {
  Object.assign(newPlan, {
    title: "",
    destination: "",
    startDate: "",
    endDate: "",
    budget: "",
    status: "planning",
    markdown: defaultMarkdown(),
    notes: "",
  })
  createMode.value = "input"
  showCreate.value = true
}

function closeCreate() {
  showCreate.value = false
}

function handleSavePlan() {
  if (!newPlan.title.trim()) {
    uni.showToast({ title: "请填写计划标题", icon: "none" })
    return
  }
  if (!newPlan.markdown.trim()) {
    uni.showToast({ title: "请填写 Markdown 内容", icon: "none" })
    return
  }
  const plan = {
    id: "user-" + Date.now(),
    title: newPlan.title,
    destination: newPlan.destination || "未填写",
    startDate: newPlan.startDate || "",
    endDate: newPlan.endDate || "",
    budget: newPlan.budget || "",
    status: newPlan.status,
    notes: newPlan.notes || "",
    contentMarkdown: newPlan.markdown,
    fromStatic: false,
  }
  plans.value.unshift(plan)
  savePlans()
  showCreate.value = false
  uni.showToast({ title: "保存成功", icon: "success" })
}

onMounted(() => {
  if (!uni.getStorageSync("isLoggedIn")) {
    uni.reLaunch({ url: "/pages/login/login" })
    return
  }
  loadPlans()
})
</script>
